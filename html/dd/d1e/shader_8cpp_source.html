145 unsigned int Shader::createShader(unsigned int type, string fpath)
146 {
148  char c;
149  string shaderCode;
150  unsigned int shaderobj;
151  FILE *shaderFile = fopen(fpath.c_str(), "r");
152  if (!shaderFile)
153  {
154  cout << "\n\n\tError opening file " << fpath << ".\n\n";
155  return 0;
156  }
158  while ((c = std::fgetc(shaderFile)) != EOF)
159  {
160  shaderCode += c;
161  }
162  fclose(shaderFile);
163  const GLchar* glShaderCode = shaderCode.c_str();
165  try
166  {
167  shaderobj = glCreateShader(type);
168  if (!shaderobj)
169  {
170  cout << "\n\n\tUnable to create the shader object.\n\n";
171  return 0;
172  }
173  else
174  {
175  cout << "\n\n\tCreated the shader object.\n\n";
176  }
177  glShaderSource(shaderobj, 1, &glShaderCode, nullptr);
178  glCompileShader(shaderobj);
180  glGetShaderiv(shaderobj, GL_INFO_LOG_LENGTH, &infoLength);
181  if(infoLength > 0)
182  {
183  char infoLog[infoLength];
184  glGetShaderInfoLog(shaderobj, infoLength, NULL, infoLog);
185  cout << "\n\nShader compilation error: \n" << infoLog << endl;
186  glDeleteShader(shaderobj);
187  return 0;
188  }
189  else
190  {
191  cout << "\n\n\tShader compiled.\n\n";
192  }
193  return shaderobj;
194  }
195  catch(exception exc)
196  {
197  cout << "\n\n\tError making shader: " << exc.what() << "\n\n";
198  return 0;
199  }

207 bool Shader::createBinary()
208 {
210  if (progLength <= 0)
211  {
212  cout << "\n\n\tShader program length less than one.\n\n";
213  progLength = 1000000;
214  }
215  binary = new unsigned char[progLength];
216  glGetProgramBinary(Program, progLength, &progLenRet, &format, (GLvoid*) binary);
217  if (progLength != progLenRet)
218  {
219  cout << "\n\n\tWarning program length of " << progLength
220  << " does not equal the size of the created binary "
221  << progLenRet << " with format " << format
222  << "\n\n";
223  }
224  FILE *shaderFile = fopen(outputFile.c_str(), "wb");
225  if (!shaderFile)
226  {
227  cout << "\n\n\tError opening file " << "shaders/cubeshader" << ".\n\n";
228  return false;
229  }
230  for (int x = 0; x < progLenRet; x++)
231  {
232  fputc(binary[x], shaderFile);
233  }
234  return true;
235 }
