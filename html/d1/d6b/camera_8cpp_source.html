134 void Camera::updateCameraVectors()
135 {
137  vec3 front;
138  front.x = cos(Yaw * onedegree) * cos(Pitch * onedegree);
139  front.y = sin(Pitch * onedegree);
140  front.z = sin(Yaw * onedegree) * cos(Pitch * onedegree);
141  Front = normalize(front);
143  Right = normalize(cross(Front, WorldUp));
144  Up = normalize(cross(Right, Front));
145 }
